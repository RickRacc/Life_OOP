231 // Allocate padded grids (+2 for top/bottom/left/right padding)
232 _prev_grid.resize(rows + 2, vector<T>(cols + 2));

241 void eval() {
242 vector<vector<int>> neighbor_counts(_rows + 2, vector<int>(_cols + 2, 0));
243
244 // Phase 1: each live cell updates its neighbors' counts
245 for (size_t i = 1; i <= _rows; ++i) {
246 for (size_t j = 1; j <= _cols; ++j) {
247 _prev_grid[i][j].update_neighbors(_prev_grid, neighbor_counts, i, j);
248 }
249 }
250
251 // Phase 2: apply evolution rules using neighbor counts
252 for (size_t i = 1; i <= _rows; ++i) {
253 for (size_t j = 1; j <= _cols; ++j) {
254 _next_grid[i][j] = _prev_grid[i][j]; // copy current cell
255 _next_grid[i][j].evolve(neighbor_counts[i][j]);
256 }
257 }
258
259 // Swap current and next generation
260 swap(_prev_grid, _next_grid);
261 }

267 _prev_grid[i + 1][j + 1] = cell;
